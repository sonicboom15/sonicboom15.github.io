package test;
public class hashValue {
int i;
public int hashCode()
{ return i;}

public boolean equals (Object obj) {
hashValue xObj = (hashValue)obj;
if (xObj.i == this.i)
return true;
else
return false;
}
public static void main(String[] args) {
hashValue h1 = new hashValue();

hashValue h2 = new hashValue();

if (h1.hashCode() == h2.hashCode())
if (h1.equals(h2))
